Select Top 1000000 tblComputers.Computer, tblSoftware.softwareName From tblComputers Inner Join    tblComputersystem On tblComputers.Computername =     tblComputersystem.Computername Left Outer Join    tblOperatingsystem On tblComputers.Computername =     tblOperatingsystem.Computername Inner Join           web40ActiveComputers On tblComputers.Computername =     web40ActiveComputers.Computername Inner Join    Web40OSName On Web40OSName.Computername = tblComputers.Computername Inner Join   tblSoftware On  tblComputers.Computername = tblSoftware.ComputerName Where (tblSoftware.softwareName Not Like '%live%' And    tblSoftware.softwareName Not Like '%connector%' And   tblSoftware.softwareName Not Like '%validation%' And    tblSoftware.softwareName Not Like '%web%' And tblSoftware.softwareName Like   '%microsoft office%') And  tblComputersystem.PartOfDomain = 1 Order By tblComputers.Computer
